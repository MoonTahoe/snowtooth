var mongoose = require('mongoose');

var newsSchema = mongoose.Schema({

    /**
     * The title for the news article
     * @property title
     * @for Models.news
     * @type {String}
     */
    title: String,

    /**
     * The url for the image that relates to the news article
     * @property img
     * @for Models.news
     * @type {String}
     */
    img: String,

    /**
     * A brief description of the news article
     * @property description
     * @for Models.news
     * @type {String}
     */
    description: String,

    /**
     * The main html content for the news article
     * @property content
     * @for Models.news
     * @type {String}
     */
    content: String,

    /**
     * The name of the person who wrote the article
     * @property author
     * @for Models.news
     * @type {String}
     */
    author: String,

    /**
     * The date that the news article was writeen
     * @property date
     * @for Models.news
     * @type {Date}
     */
    date: Date

});

/**
 * A data model containing the snowtooth news collection
 * @uses mongoose
 * @class Models.news
 */
module.exports.news = mongoose.model('news', newsSchema, 'news');
