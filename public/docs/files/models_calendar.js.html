var mongoose = require('mongoose');

var calendarSchema = mongoose.Schema({

    /**
     * A brief unique string id for for the event to be \
     * displayed on the home page
     * @property id
     * @for Models.calendar
     * @type {String}
     */
    id: String,

    /**
     * A title for the event
     * @property title
     * @for Models.calendar
     * @type {String}
     */
    title: String,

    /**
     * An image for the event
     * @property img
     * @for Models.calendar
     * @type {String}
     */
    img: String,

    /**
     * The date and time that the event begins
     * @property start
     * @for Models.calendar
     * @type {Date}
     */
    start: Date,

    /**
     * The date and time that the event ends
     * @property end
     * @for Models.calendar
     * @type {Date}
     */
    end: Date,

    /**
     * A description of the event
     * @property description
     * @for Models.calendar
     * @type {String}
     */
    description: String,

    /**
     * Any HTML content that is associated with the event
     * @property html
     * @for Models.calendar
     * @type {String}
     */
    html: String

});

/**
 * A data model containing the snowtooth events calendar collection
 * @uses mongoose
 * @class Models.calendar
 */
module.exports.calendar = mongoose.model('calendar', calendarSchema, 'calendar');
